CREATE TABLE "admin" (
	"user_id"	TEXT NOT NULL,
	"description"	TEXT NOT NULL,
	PRIMARY KEY("user_id")
);
CREATE TABLE "category" (
	"category"	TEXT NOT NULL UNIQUE,
	"description"	TEXT NOT NULL,
	"date"	TEXT NOT NULL,
	PRIMARY KEY("category")
);
CREATE TABLE "influencer" (
	"username"	TEXT UNIQUE,
	"address"	TEXT,
	"contact"	NUMERIC,
	"district"	TEXT,
	"state"	TEXT,
	"pincode"	NUMERIC,
	"first_name"	TEXT,
	"last_name"	TEXT,
	"instagram_id"	TEXT,
	"linkedin_id"	TEXT,
	"facebook_id"	TEXT,
	"x_id"	TEXT,
	"category"	TEXT,
	"insta_f"	INTEGER,
	"linkedin_f"	INTEGER,
	"facebook_f"	INTEGER,
	"x_f"	INTEGER,
	"bio"	TEXT,
	"flag"	INTEGER,
	"wallet"	INTEGER,
	"complete"	INTEGER,
	PRIMARY KEY("username"),
	FOREIGN KEY("category") REFERENCES "category"("category")
);
CREATE TABLE role (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(80) UNIQUE NOT NULL,
    description VARCHAR(255)
);
CREATE TABLE "sponsor" (
	"username"	TEXT UNIQUE,
	"address"	TEXT,
	"contact"	TEXT,
	"district"	TEXT,
	"state"	TEXT,
	"pincode"	INTEGER,
	"first_name"	TEXT,
	"last_name"	TEXT,
	"company_name"	TEXT,
	"industry"	TEXT,
	"positions"	TEXT,
	"bio"	TEXT,
	"flag"	INTEGER,
	"wallet"	INTEGER complete INTEGER,
	"category"	TEXT,
	FOREIGN KEY("category") REFERENCES "category"("category"),
	PRIMARY KEY("username")
);
CREATE TABLE user (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255),
    active BOOLEAN,
    fs_uniquifier VARCHAR(65) UNIQUE NOT NULL
);
CREATE TABLE user_roles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    role_id INTEGER NOT NULL,
    FOREIGN KEY (user_id) REFERENCES user (id) ON DELETE CASCADE,
    FOREIGN KEY (role_id) REFERENCES role (id) ON DELETE CASCADE
)
